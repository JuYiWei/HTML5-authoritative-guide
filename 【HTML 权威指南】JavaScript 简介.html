一、准备 JavaScript

1. 内嵌 JavaScript

<script>
    document.write("hello world");
</script>

2. JavaScript 文件

  • src 指向文件路径
<script src="//cdn.bootcss.com/jquery/3.2.1/jquery.min.js"></script>

二、使用语句

  • 略；

三、函数的定义与使用

<script>
    function testSay(name, content) {
        return 'hey！' + name + ':' + content;
    }
    document.write(testSay('LiLei','早上好'));
</script>

四、变量与类型

1. 定义变量

var name = "Li Lei";

2. 基本类型

  • 字符串 string：单/双引号都可以，但是前后要匹配；
  • 布尔 boolean：ture/false；
  • 数值 number：整数/浮点；

3.创建对象

var aObj = {
    // 属性
    name:"Li Lei",
    address:"hang zhou",

    // 函数
    myLog:function () {
        document.write(this.name + " welcome to " + this.address)
    }
};

// 使用属性
document.write(aObj.name+aObj.address);

// 使用函数
aObj.myLog();

4. 使用对象

  • 读取修改属性（属性其实是在对象的属性列表，数组形式）
aObj.name = "Han meimei";
aObj["name"] = "Xiao Ming";

document.write(aObj.name);
  • 枚举对象属性
for (var prop in aObj) {
  document.write(prop+" ");
}
  • 增删属性和方法
// 增
aObj.age = 18;
aObj.say = function () {
    document.write("Hello");
};

// 删
delete aObj.address;
delete aObj.say;
  • 判断是否有某属性
var hasAge = "age" in aObj;
document.write(hasAge);

五、运算符

1. 相等和等同

  • 相等（不判断类型，只判断值）
var a = 1;
var b = "1";
if (a==b) {
   document.write("same");
}
  • 等同（判断类型）
var a = 1;
var b = "1";
if (a===b) {
   document.write("same");
}

2. 显式类型转换

  • 数值转字符串
var a = (18).toString();
var b = String(18);
方法 说明
toString() 十进制
toString(2) 二进制
toString(8) 八进制
toString(16) 十六进制
toFixed(n) 保留 n 位小数
toExpontial(n) 指数表示，n 位小数
toPrecision(n) n 位有效数字
  • 字符串转数值
var a = Number("12");
方法 说明
Number() 整数或实数
parseInt() 整数
parseFloat() 整数或实数

六、数组

  • 创建
var array = [];
array[0] = "Li Lei";
  • 数组字面量
var array = ["Li Lei",18,"hangzhou"];
  • 读取修改
array[0] = "Han Meimei";
document.write(array[0]);
  • 枚举
for (var i=0;i<array.length;i++) {
   document.write(array[i]);
}
  • 内置数组方法
方法 说明 返回
concat(arr) 拼接数组 数组
join(sep} 数组转字符串，使用 sep 分隔 字符串
pop() 删除并返回最后一个 对象
push(i) 末尾添加一个数据 void
reverse() 反转顺序 数组
shift 删除并返回第一个 对象
slice(s,e) 子数组 数组
sort() 排序 数组
unshift(i) 删除第一个 void

七、处理错误

1. try - catch

try {
    var myArray;
    for (var i=0;i<3;i++) {
        document.write(myArray[i]);
    }
} catch (e) {
    document.write("error" + e);
}

2. Error 对象

属性 说明 返回
message 说明 字符串
name 错误名称 字符串
number 错误代码 数值

八、undefined 和 null

undefined：未赋值或无对象属性；
null：已赋值，但值不是有效类型；

1. 检查

  • 都可以直接用逻辑非!判断；
var a;
var b = null;

if (!a) {
    document.write("undefined ");
}

if (!b) {
    document.write("null ");
}

2. 区分

  • undefined 与 null，值相等，但类型不一致；所以用相等于等同来区分；
var a;
var b = null;

if (a==b) {
    document.write("相等 ");
}

if (a===b) {
    document.write("等同 ");
}
