5.4 使用 canvas 元素（1）

一、开始使用 canvas 元素

<canvas width="200" height="100" style="background: gray">
    你的浏览器不支持 canvas；
</canvas>

二、获取画布的上下文

属性 说明
heigth
width
getContext() 画布上下文 
<canvas id="myCanvas" width="200" height="100" style="background: gray">
    你的浏览器不支持 canvas；
</canvas>

<script>
    var ctx = document.getElementById("myCanvas").getContext("2d");
    ctx.fillRect(10,20,30,40);
</script>

三、绘制矩形

方法 说明
clearRect(x,y,w,h) 清除
fillRect(x,y,w,h) 填充
stockRect(x,y,w,h) 描边 
<canvas id="myCanvas" width="200" height="100" style="background: gray">
    你的浏览器不支持 canvas；
</canvas>

<script>
    var ctx = document.getElementById("myCanvas").getContext("2d");
    ctx.fillRect(10,20,30,40);
    ctx.strokeRect(10,20,40,50);
    ctx.clearRect(10,20,10,10);
</script>

四、设置画布绘制状态

属性 说明
fillStyle 填充样式
lineJoin 连接样式
lineWidth 线宽
strokeStyle 描边样式

1. 线条连接样式

    ctx.lineJoin = "round";
    ctx.lineJoin = "bevel";
    ctx.lineJoin = "miter";

2. 填充与描边样式

    ctx.fillStyle = "red";
    ctx.strokeStyle = "rgb(255,0,255)";

3. 线性渐变（x,y,w,h）

    var c = ctx.createLinearGradient(10,20,30,40);
    c.addColorStop(0,"red");
    c.addColorStop(0.5,"white");
    c.addColorStop(1,"green");

    ctx.fillStyle = c;
    ctx.fillRect(10,20,30,40);

4. 径向渐变（x1,y1,r1,x2,y2,r2）

    var c = ctx.createRadialGradient(100,100,100,100,100,50);
    c.addColorStop(0,"red");
    c.addColorStop(0.5,"white");
    c.addColorStop(1,"green");
    ctx.fillStyle = c;
    ctx.fillRect(0,0,200,200);

5. 使用图案

值 说明
repeat 水平垂直重复
repeat-x 水平重复
repeat-y 垂直重复
no-repeat 不重复 
    var img = document.getElementById("img");
    var pattetn = ctx.createPattern(img,"repeat");

    ctx.fillStyle = pattetn;
    ctx.fillRect(0,0,200,200);

五、保存和恢复绘制状态

值 睡眠
save() 添加保存一次
restore() 上一次的保存 
    var index = 0;
    var arr = ["red","black","green"];
    var ctx = document.getElementById("myCanvas").getContext("2d");

    function save() {
        ctx.save();
        ctx.fillStyle = arr[index%3];
        index++;
        ctx.fillRect(0,0,200,200);
    }

    function restore() {
        ctx.restore();
        ctx.fillRect(0,0,200,200);
    }

六、绘制图像

1. drawImage(...)，有 3、5、9 参数，分别代表不同意思；

三参数：2、3 代表图像在画布的位置；
五参数：4、5 代表图像宽高；
九参数：2、3 代表原图内偏移，4、5 原图使用的宽高，6、7 图像在画布的位置，8、9 图像宽高；

2. 使用视频图像

<img id="img" hidden src="src/test_on.png">

<video autoplay width="200px" height="200px" id="video" src="src/mp4.mp4">??</video>
<button onclick="save()">save</button>
<script>
    var ctx = document.getElementById("myCanvas").getContext("2d");

    function save() {
        var img = document.getElementById("video");
        ctx.drawImage(img,0,0,200,200);
    }
</script>

3. 使用画布图像

<canvas id="myCanvas" width="200" height="200" style="background: gray">
    你的浏览器不支持 canvas；
</canvas>

<canvas id="myCanvas2" width="200" height="200" style="background: gray">
    你的浏览器不支持 canvas；
</canvas>

<script>
    var ctx = document.getElementById("myCanvas").getContext("2d");
    ctx.fillRect(0,0,100,100);

    var ctx2 = document.getElementById("myCanvas2").getContext("2d");
    var ctxImg = document.getElementById("myCanvas");
    ctx2.drawImage(ctxImg,0,0,100,100);
    ctx2.strokeRect(0,0,100,100);
</script>
